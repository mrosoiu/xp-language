  select
    e.event_id as id,
    e.title,
    e.description,
    l.name as location
  from
    event e,
    location l
  where l.location_id = e.location_id
    and e.bz_id= 20000
</pre>

<code>
  $c= create(new Criteria())
    ->on(Event::getPeer())
    ->join(Location::getPeer(), 'location_id')
    ->add('bz_id', 20000, EQUAL)
  ;
</code>
